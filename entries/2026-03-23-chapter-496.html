// Water Breathing Spell
class WaterBreathing {
    constructor(caster) {
        this.caster = caster;
        this.duration = 7200; // 2時間
        this.targets = [];
    }
    
    apply(target) {
        target.status.underwaterBreathing = true;
        target.status.duration = this.duration;
        this.targets.push(target);
        return {
            status: "applied",
            remainingTime: this.duration
        };
    }
    
    applyAll(party) {
        party.members.forEach(member => {
            this.apply(member);
        });
        return {
            affectedCount: party.members.length,
            duration: this.duration
        };
    }
}

// 罠回避システム
class TrapAvoidance {
    scan(tiles) {
        const safePath = [];
        tiles.forEach((row, y) => {
            row.forEach((tile, x) => {
                if (tile.pressure === 0) {
                    safePath.push({ x, y });
                }
            });
        });
        return {
            safeTiles: safePath,
            dangerTiles: tiles.length * tiles[0].length - safePath.length,
            instructions: "安全なタイルのみ踏んで進む"
        };
    }
    
    guide(party, path) {
        return {
            steps: path.length,
            warning: "足元を常に確認",

        };

// Code Breaker Spell
class CodeBreaker {
    constructor(target) {
        this.target = target;
        this.effect = "防御バフ解除";
        this.duration = 10; // 秒
    }
    
    execute() {
        // 敵の防御システムをハッキング
        this.target.buffs.forEach(buff => {
            if (buff.type === "defense") {
                buff.active = false;
            }
        });
        return {
            status: "success",
            removedBuffs: ["Dark Barrier"],
            vulnerabilityDuration: this.duration
        };
    }
}

const codeBreaker = new CodeBreaker(templar);
codeBreaker.execute();
