class WarningDecoder {
    constructor() {
        this.language = 'ancient_magic';
        this.translator = new AncientTranslator();
    }
    
    decode(inscription) {
        const raw = this.extractText(inscription);
        const translated = this.translator.toModern(raw);
        
        return {
            original: raw,
            translation: translated,
            severity: this.assessSeverity(translated)
        };
    }
    
    assessSeverity(text) {
        const dangerKeywords = ['death', 'forbidden', 'destruction', 'soul'];
        const count = dangerKeywords.filter(k => 
            text.toLowerCase().includes(k)
        ).length;
        
        if (count >= 3) return 'CRITICAL';
        if (count >= 2) return 'HIGH';
        if (count >= 1) return 'MEDIUM';
        return 'LOW';
    }
}

class SourceController {
    constructor() {
        this.manaFlow = new ManaFlowSystem();
        this.balance = new BalanceCalculator();
    }
    
    adjustFlow(params) {
        const { consumption, regeneration, threshold } = params;
        
        // 現在のバランスを計算
        const current = this.balance.calculate();
        
        // 目標バランスを設定
        const target = {
            consumption: consumption || current.consumption * 0.8,
            regeneration: regeneration || current.regeneration * 1.2,
            threshold: threshold || 0.7
        };
        
        // 調整を実行
        return this.manaFlow.adjust(current, target);
    }
    
    monitor() {
        setInterval(() => {
            const status = this.balance.calculate();
            
            if (status.ratio < 0.5) {
                this.triggerAlert('Mana depletion warning');
            }
            
            if (status.ratio > 0.9) {
                this.triggerAlert('Mana overflow warning');
            }
        }, 60000); // 1分ごと
    }
    
    emergencyOverride() {
        // 緊急時の安全策
        this.manaFlow.emergencyStop();
        this.backupData();
        this.notifyAdministrators();
    }
}
