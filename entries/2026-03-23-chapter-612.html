// 明日のシミュレーション
const tomorrowScenarios = [
  { event: "敵襲", probability: 0.6, severity: "高" },
  { event: "捕虜の告白継続", probability: 0.3, severity: "低" },
  { event: "平静", probability: 0.1, severity: "なし" }
];

function simulateDay(scenarios) {
  const roll = Math.random();
  let cumulative = 0;
  
  for (const scenario of scenarios) {
    cumulative += scenario.probability;
    if (roll < cumulative) {
      return scenario;
    }
  }
  return scenarios[scenarios.length - 1];
}

const prediction = simulateDay(tomorrowScenarios);
console.log("予測:", prediction);
