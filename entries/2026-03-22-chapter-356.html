// Activate Compass
// 羅針盤起動

function activateCompass(compass: SpaceTimeCompass, user: User): Path | null {
  // Check requirements
  const requirements = compass.requirements;
  const userResistances = user.resistances;
  
  const hasAllResistances = requirements.every(req => 
    userResistances.includes(req)
  );
  
  if (!hasAllResistances) {
    console.log("耐性が不足しています");
    return null;
  }
  
  // Find path
  const path: Path = {
    destination: "時空の神殿",
    steps: [],
    estimatedTime: "不明"
  };
  
  // The path reveals itself through meditation
  console.log("道が開かれました");
  return path;
}

// Temple Access Check
// 神殿アクセス確認

interface AccessCheck {
  user: string;
  resistances: string[];
  accessGranted: boolean;
}

function checkTempleAccess(user: User): AccessCheck {
  const requiredResistances = [
    "ice",
    "spacetime", 
    "light",
    "darkness"
  ];
  
  const userResistances = user.resistances;
  const hasAllResistances = requiredResistances.every(
    res => userResistances.includes(res)
  );
  
  return {
    user: user.name,
    resistances: userResistances,
    accessGranted: hasAllResistances
  };
}
