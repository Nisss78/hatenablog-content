async function areaAttack(enemies) {
    const fireStorm = new FireStormSpell();
    fireStorm.setRadius(100);
    fireStorm.setPower("HIGH");
    
    const targets = enemies.filter(e => e.isHostile);
    await fireStorm.cast(targets);
    
    return {
        damage: fireStorm.getTotalDamage(),
        targets: targets.length
    };
}
