// 戦況分析
class BattleAnalyzer {
    constructor() {
        this.allies = [
            {name: "Kaito", power: 85, element: "wind", status: "active"},
            {name: "Elina", power: 70, element: "fire", status: "active"},
            {name: "Liam", power: 65, element: "none", status: "injured"},
            {name: "Cecilia", power: 60, element: "water", status: "active"},
            {name: "Mina", power: 0, element: "unknown", status: "awakening"}
        ];
        this.enemies = [
            {name: "Enemy1", power: 80, element: "dark", status: "active"},
            {name: "Enemy2", power: 75, element: "dark", status: "active"},
            {name: "Enemy3", power: 75, element: "dark", status: "active"},
            {name: "Enemy4", power: 70, element: "dark", status: "active"},
            {name: "Enemy5", power: 70, element: "dark", status: "active"}
        ];
    }

    calculateWinRate() {
        const allyPower = this.allies.reduce((sum, a) => sum + a.power, 0);
        const enemyPower = this.enemies.reduce((sum, e) => sum + e.power, 0);
        return Math.round((allyPower / enemyPower) * 100);
    }
}

const analyzer = new BattleAnalyzer();
const winRate = analyzer.calculateWinRate(); // 勝率約52%
