// Ancient Inscription Analysis
// Language: Pre-Magic Era Script
// Translation algorithm:

function translateInscription(text) {
  const patterns = {
    "→": "forward",
    "←": "backward", 
    "↑": "ascend",
    "↓": "descend"
  };
  
  return text.split("").map(c => patterns[c] || c).join("");
}

// Result: "→ → ↑ ← ↓" 
// Meaning: Forward, Forward, Up, Back, Down

// Analysis: Elemental Balance
// Purpose: Find pattern

function analyzeElementalWall() {
  const elements = {
    fire: { size: 100, position: 1 },
    water: { size: 25, position: 2 },
    wind: { size: 25, position: 3 },
    earth: { size: 25, position: 4 }
  };
  
  // Calculate ratio
  const total = Object.values(elements).reduce((sum, e) => sum + e.size, 0);
  
  return {
    firePercent: (elements.fire.size / total) * 100,
    othersPercent: ((total - elements.fire.size) / total) * 100,
    hint: "Fire dominates, but others exist. Balance is key."
  };
}

analyzeElementalWall();
