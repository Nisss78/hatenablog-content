// Guardian Assignment - 守護者任務割り当て
// 各依頼を適切な守護者に割り当てる

function assignGuardians(requests) {
  const assignments = [];
  
  requests.forEach(request => {
    const guardian = selectGuardian(request.region, request.type);
    
    const assignment = {
      requestId: request.id,
      guardianId: guardian.id,
      task: request.task,
      location: request.location,
      reward: request.reward,
      priority: request.priority
    };
    
    // 守護者に通知
    notifyGuardian(guardian.id, assignment);
    
    assignments.push(assignment);
  });
  
  return assignments;
}

// 各依頼の割り当て
const requests = [
