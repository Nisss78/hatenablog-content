Water_Domain


function Water_Domain(radius, density) {
    const waterParticles = [];
    
    for (let i = 0; i < density; i++) {
        waterParticles.push({
            x: Math.random() * radius,
            y: Math.random() * radius,
            z: Math.random() * radius,
            state: "liquid",
            control: this.mastery
        });
    }
    
    return {
        domain: waterParticles,
        controlRadius: radius,
        absorptionRate: 0.85,
        pressure: this.magicPower * 2
    };
}

Ice_vs_Fire


function ElementalCollision(ice, fire) {
    const icePower = ice.temperature * ice.density;
    const firePower = fire.temperature * fire.intensity;
    
    if (icePower > firePower) {
        return {
            result: "ice_dominant",
            sideEffect: "steam_generation",
            residualTemperature: ice.temperature
        };
    } else if (firePower > icePower) {
        return {
            result: "fire_dominant",
            sideEffect: "melting",
            residualTemperature: fire.temperature
        };
    } else {
        return {
            result: "stalemate",
            sideEffect: "massive_steam",
            residualTemperature: 373 // 蒸気
        };
    }
}

Oxygen_Denial


function OxygenDenial(target, radius) {
    // 対象周囲の酸素を排除
    const atmosphere = getAtmosphere(target.position, radius);
    
    atmosphere.filter(molecule => molecule.type === "O2").forEach(oxygen => {
        oxygen.displace("N2"); // 窒素で置換
    });
    
    return {
        oxygenLevel: 0,
        duration: 30,
        targetSuffocation: true,
        fireSuppression: "instant"
    };
}
