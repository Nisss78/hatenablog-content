function repairSouthAnchor() {
    const anchor = getAnchor("south");
    const bow = getArtifact("windGodBow");
    
    // Fill cracks in core
    anchor.core.cracks.forEach(crack => {
        const windEnergy = bow.channelMana("wind");
        crack.fill(windEnergy);
    });
    
    // Restore rune patterns
    anchor.runes = restorePattern("wind", anchor.runes);
    
    // Optimize wind flow
    anchor.optimizeFlow();
    
    return anchor.restore();
}
