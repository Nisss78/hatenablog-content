// 集団転移の実行
class GroupTeleport extends AdminMagic {
    constructor(party, destination) {
        super();
        this.party = party;
        this.destination = destination;
        this.method = "座標置換";
    }
    
    linkMembers() {
        return this.party.map(member => {
            return {
                name: member.name,
                currentLocation: member.getCoordinates(),
                linked: true
            };
        });
    }
    
    execute() {
        const linked = this.linkMembers();
        const source = this.getCurrentSpace();
        const target = this.getDestinationSpace();
        
        // 空間の交換
        source.swap(linked, target);
        
        return {
            status: "転移完了",
            destination: this.destination.name,
            partySize: linked.length
        };
    }
}

// 実行
const teleport = new GroupTeleport(
    ["カイト", "エリナ", "リアム", "ヒナタ", "セシリア"],
    worldTreeLocation
);
teleport.execute();
