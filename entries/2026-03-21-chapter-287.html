    constructor(target, duration) {
        super();

class SealReinforcement extends TimeSpaceMagic {
    constructor(sealId) {

        this.target = sealId;
        this.method = "パッチ適用";
    }

        const patch = this.createTemporalBarrier();
        patch.cover(crack);
