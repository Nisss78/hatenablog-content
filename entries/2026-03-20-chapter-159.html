class MentalShield {
    constructor(targets) {
        this.targets = targets;
        this.strength = 100;
        this.duration = "UNLIMITED";
    }
    
    async cast() {
        for (const target of this.targets) {
            const shield = new PsychicBarrier();
            shield.setResistance("FEAR", "MAX");
            shield.setResistance("ILLUSION", "MAX");
            shield.setResistance("MIND_CONTROL", "MAX");
            await shield.applyTo(target);
        }
        return { status: "SUCCESS", protected: this.targets.length };
    }
}

const party = ["Kaito", "Elina", "Liam", "Gantz", "Cecilia", "Oscar", "Herman"];
const shield = new MentalShield(party);
shield.cast();

class NightmareRescue {
    constructor(target) {
        this.target = target;
        this.dreamAnchor = new SoulAnchor();
    }
    
    async enterDreams() {
        // 自分の魂を守りながら悪夢に入る
        const protection = new SelfProtection();
        protection.activate();
        
        // 悪夢の世界への侵入
        const dreamRealm = await this.dreamAnchor.connect(this.target);
        return dreamRealm;
    }
    
    async rescueSoul() {
        const dream = await this.enterDreams();
        const trappedSoul = dream.findSoul();
        
        // 魂を現実世界に導く
        await this.dreamAnchor.extract(trappedSoul);
        
        return { status: "RESCUED", soul: trappedSoul };
    }
}

const rescue = new NightmareRescue("statue_001");
rescue.rescueSoul();
