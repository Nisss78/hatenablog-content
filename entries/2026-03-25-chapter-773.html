// 闇との対話
function communicateWithDarkness() {
  // 心を静める
  const calmMind = meditation(depth);
  
  // 闇に問いかける
  const question = "何を知っている？";
  const response = darkness.whisper(calmMind, question);

    heard: true,
    content: response,
    type: "静かな囁き"
  };
}
