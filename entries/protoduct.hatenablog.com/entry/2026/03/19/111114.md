// Communication Network System
// Uses magic circles as network nodes

const Network = {
  nodes: [],
  
  addNode(circle) {
    circle.id = generateId();
    this.nodes.push(circle);
    return circle.id;
  },
  
  connect(node1Id, node2Id) {
    const node1 = this.nodes.find(n => n.id === node1Id);
    const node2 = this.nodes.find(n => n.id === node2Id);
    
    // Create magical link
    const link = createLink(node1, node2);
    link.establish();
    return link;
  },
  
  send(fromId, toId, message) {
    const link = this.findLink(fromId, toId);
    if (!link) {
      return "Error: No connection";
    }
    
    const packet = this.encode(message);
    link.transmit(packet);
    return "Message sent";
  },
  
  encode(message) {
    return message.split('').map(c => ({
      char: c,
      magic: convertToMagic(c)
    }));
  }
};

// Create communication spell
function comm(targetId, message) {
  return Network.send(myId, targetId, message);
