// 敵感知
class EnemyDetector {
    scan(area) {
        const result = {
            enemies: 0,
            type: null,
            distance: null
        };
        
        // 森の中をスキャン
        for (let entity of area.entities) {
            if (entity.alignment === "hostile") {
                result.enemies++;
                result.type = entity.type;
                result.distance = entity.distance;
            }
        }
        
        return result;
    }
}
