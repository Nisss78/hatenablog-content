// Spell: Thermal Shield
// Class: Barrier magic
// Element: Water/Fire hybrid
// Purpose: Fire resistance
// Code:

function createThermalShield(target) {
  const orb = getAquaOrb();
  const water = orb.extractEssence();
  const fire = getEnvironment().fireEnergy;
  
  // Balance water and fire
  const balance = water.mix(fire, { ratio: 0.5 });
  
  target.applyShield({
    type: "thermal_barrier",
    resistance: 90,
    duration: 3600,
    source: balance
  });
  
  return `Shield applied to ${target.name}`;
}

// Apply to party
party.members.forEach(m => createThermalShield(m));
