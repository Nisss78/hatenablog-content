function analyzeWindPattern(windStream) {
  const rhythm = [];
  let currentValue = 0;
  
  for (let i = 0; i < windStream.length; i++) {
    const delta = windStream[i] - currentValue;
    if (Math.abs(delta) > THRESHOLD) {
      rhythm.push({
        timing: i,
        direction: delta > 0 ? "rise" : "fall",
        magnitude: Math.abs(delta)
      });
    }
    currentValue = windStream[i];
  }
  
  return rhythm;
}
