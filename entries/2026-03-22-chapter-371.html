// 通信リンク

interface CommLink {
    channel: string;
    range: number;
    encryption: boolean;
    batteryLife: number;
}

    const link: CommLink = {
        channel: "PARTY_CHANNEL_07",
        range: 5000,            // meters
        encryption: true,
        batteryLife: 480        // minutes
    };
