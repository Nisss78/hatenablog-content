        this.position = position;

        this.phase = "初期化";

    }

    async phaseOne_Foundation() {
        this.phase = "基盤構築";
        console.log("--- フェーズ1: 基盤構築 ---");
        
        // 地脈との接続確立
        const leylineConnection = await this.establishLeylineConnection();
        console.log(`地脈接続: ${leylineConnection.status}`);
        
        // 水脈との同期
        const aquiferSync = await this.syncWithAquifer();
        console.log(`水脈同期: ${aquiferSync.strength}%`);
        
        this.energyLevel += 3000;
    }

    async phaseTwo_ElementInjection() {
        this.phase = "属性注入";
        console.log("--- フェーズ2: 属性注入 ---");
        
        // 風属性の注入
        const windInjection = await this.injectElement("wind");
        console.log(`風属性注入量: ${windInjection.amount}`);
        
        // 水属性の注入
        const waterInjection = await this.injectElement("water");
        console.log(`水属性注入量: ${waterInjection.amount}`);
        
        // 属性比率の調整（水60%、風40%）
        const balance = this.adjustElementBalance(0.6, 0.4);
        console.log(`属性バランス: 水${balance.water * 100}%, 風${balance.wind * 100}%`);
        
        this.energyLevel += 5000;
    }

    async phaseThree_ResonanceSync() {
        this.phase = "共鳴同期";
        console.log("--- フェーズ3: 共鳴同期 ---");
        
        // 風と水の共鳴周波数を探索
        const resonanceFreq = await this.findResonanceFrequency();
        console.log(`共鳴周波数: ${resonanceFreq}Hz`);
        
        // 他の頂点との同期
        const vertexSync = await this.syncWithOtherVertices();
        console.log(`頂点間同期: ${vertexSync.connected.length}個の頂点と接続`);
        
        this.energyLevel += 4000;
    }

    async phaseFour_Stabilization() {
        this.phase = "安定化";
        console.log("--- フェーズ4: 安定化 ---");
        
        // エネルギーフローの安定化
        let stabilityCounter = 0;
        while (stabilityCounter < 100) {
            const fluctuation = this.measureFluctuation();
            if (fluctuation < 0.05) {
                stabilityCounter += 5;
            } else {
                this.compensateFluctuation(fluctuation);
            }
            await this.delay(100);
        }
        
        this.stability = 95;
        this.energyLevel = this.maxEnergy;
        console.log(`安定性: ${this.stability}%`);
    }

// 結界全体の脆弱性スキャン
async function scanBarrierVulnerabilities() {
    console.log("結界脆弱性スキャン開始...\n");
    
    const vulnerabilities = [];
    
    // 各頂点間の接続をチェック
    const connections = [
        ["wind", "water"],
        ["water", "lightning"],
        ["lightning", "ice"],
        ["ice", "fire"],
        ["fire", "wind"]
    ];
    
    for (const [v1, v2] of connections) {
        const strength = await checkConnectionStrength(v1, v2);
        console.log(`${v1} - ${v2} 接続強度: ${strength}%`);
        
        if (strength < 85) {
            vulnerabilities.push({
                type: "weakConnection",
                location: `${v1} - ${v2}`,
                strength: strength,
                recommendation: "エネルギー補充が必要"
            });
        }
    }
    
    // 核との接続をチェック
    for (const vertex of ["wind", "fire", "lightning", "ice", "water"]) {
        const coreLink = await checkCoreLink(vertex);
        console.log(`${vertex}頂点 - 核 接続: ${coreLink.status}`);
        
        if (coreLink.latency > 100) {
            vulnerabilities.push({
                type: "highLatency",
                location: `${vertex} - core`,
                latency: coreLink.latency,
                recommendation: "同期調整が必要"
            });
        }
    }
    
    // 外部からの侵入可能性をシミュレーション
    const intrusionTests = await simulateIntrusions();
    console.log(`\n侵入テスト結果: ${intrusionTests.blocked}/${intrusionTests.total} ブロック`);
    
    if (vulnerabilities.length === 0) {
        console.log("\n脆弱性は検出されませんでした。結界は健全です。");
    } else {
        console.log(`\n${vulnerabilities.length}件の脆弱性を検出:`);
        vulnerabilities.forEach((v, i) => {
            console.log(`${i + 1}. ${v.type} @ ${v.location}`);
            console.log(`   推奨: ${v.recommendation}`);
        });
    }
    
    return vulnerabilities;
}
