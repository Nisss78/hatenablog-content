function Wind_Amplification(target, range) {
    // 風の拡大
    const baseWind = target.getWindAffinity();
    const boost = this.windPower * 1.5;
    
    return {
        enhancedRange: range * boost,
        duration: 3600, // 1時間
        sensitivity: baseWind * 2
    };
}

Wind_Reconnaissance


function Wind_Reconnaissance(radius) {
    const windSense = this.getEnhancedWindSense();
    const targets = [];
    
    // 風を通じて感知
    for (let direction = 0; direction < 360; direction++) {
        const scan = windSense.project(direction, radius);
        
        scan.forEach(point => {
            if (point.hasHostilePresence()) {
                targets.push({
                    direction: direction,
                    distance: point.distance,
                    signature: point.energySignature,
                    count: point.entityCount
                });
            }
        });
    }
    
    return {
        scanTime: Date.now(),
        radius: radius,
        hostiles: targets
    };
}
