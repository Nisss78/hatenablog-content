// Essence Fusion - エッセンス融合
// 七属性のエッセンスを融合させる

function fuseEssence() {
  const essences = [
    { element: "fire", crystal: "炎の結晶" },
    { element: "water", crystal: "水の結晶" },
    { element: "wind", crystal: "風の結晶" },
    { element: "earth", crystal: "地の結晶" },
    { element: "ice", crystal: "氷の結晶" },
    { element: "light", crystal: "光の結晶" },
    { element: "dark", crystal: "闇の結晶" },
    { element: "thunder", crystal: "雷の結晶" }
  ];
  
  // 融合プロセス
  const fusedEssence = essences.reduce((acc, essence) => {
    return merge(acc, essence.crystal);
  }, null);
  
  // 七色の融合結晶が完成
  return {
    name: "七属性融合結晶",
    power: "ancient",
    stability: 95
  };
}

// Primal Light Infusion - 原初の光注入
// 融合結晶に原初の光を注入する

  const infused = {
    crystal: fusedCrystal,
