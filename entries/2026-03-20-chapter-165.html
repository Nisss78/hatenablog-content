class NightmareDetector {
    constructor() {
        this.range = 100;
        this.sensitivity = "high";
    }
    
    async scan() {
        // 周囲の悪夢を検知
        const entities = await this.detectNightmareEntities();
        const threats = await this.assessThreats(entities);
        
        return {
            count: entities.length,
            threatLevel: threats.level,
            types: entities.map(e => e.type)
        };
    }
    
    async detectNightmareEntities() {
        // 悪夢の実体を検知
        return [
            { type: "shadow_crawler", threat: "moderate" },
            { type: "shadow_crawler", threat: "moderate" },
            { type: "fear_whisperer", threat: "high" }
        ];
    }
    
    async assessThreats(entities) {
        // 脅威レベルを評価
        const totalThreat = entities.reduce((sum, e) => {
            const values = { low: 1, moderate: 2, high: 3 };
            return sum + values[e.threat];
        }, 0);
        
        return {
            level: totalThreat > 5 ? "critical" : "moderate",
            score: totalThreat
        };
    }
}

const detector = new NightmareDetector();
const scan = await detector.scan();

class FearWhisperer extends NightmareEntity {
    constructor() {
        this.type = "fear_whisperer";
        this.abilities = ["fear_amplification", "mental_assault"];
        this.weakness = "light_magic";
    }
    
    async attack(target) {
        // 恐怖を増幅して攻撃
        const fear = await this.extractFear(target);
        await this.amplifyFear(target, fear);
        await this.mentalAssault(target);
        
        return {
            damage: "psychic",
            severity: "moderate"
        };
    }
    
    async extractFear(target) {
        // 対象の恐怖を抽出
        return {
            source: target.deepestFear,
            intensity: "high"
        };
    }
    
    async amplifyFear(target, fear) {
        // 恐怖を増幅
        return {
            effect: "fear_amplification",
            duration: "temporary"
        };
    }
}

class MentalShield {
    constructor(caster) {
        this.caster = caster;
        this.strength = "high";
        this.duration = 30000; // 30秒
    }
    
    async activate() {
        // 精神シールドを展開
        const barrier = await this.constructBarrier();
        const reinforcement = await this.reinforceMind();
        
        return {
            status: "active",
            protection: "psychic_immunity"
        };
    }
    
    async constructBarrier() {
        // 精神バリアを構築
        return {
            type: "mental_barrier",
            layers: 3
        };
    }
    
    async reinforceMind() {
        // 精神を強化
        return {
            type: "mental_reinforcement",
            duration: this.duration
        };
    }
}

class LightPurification {

        // 聖なる光を召喚
