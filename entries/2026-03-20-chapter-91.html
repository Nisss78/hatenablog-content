class CrystalMemoryReader {
    constructor(crystal) {
        this.crystal = crystal;
        this.encryption = crystal.getEncryption();
        this.memories = [];
    }
    
    decrypt() {
        const key = this.generateDecryptionKey();
        const encrypted = this.crystal.getRawData();
        
        try {
            const decrypted = this.applyHarmonics(encrypted, key);
            this.memories = this.parseMemories(decrypted);
            return true;
        } catch (e) {
            console.error('Decryption failed:', e);
            return false;
        }
    }
    
    generateDecryptionKey() {
        // ハーモニーシステムの共鳴を鍵として使用
        const harmony = HarmonySystem.getInstance();
        const resonance = harmony.getCurrentResonance();
        
        return {
            frequency: resonance.baseFrequency,
            amplitude: resonance.amplitude,
            phase: resonance.phase
        };
    }
    
    parseMemories(data) {
        const parser = new MemoryParser();
        return parser.parse(data);
    }
    
    play(memoryId) {
        const memory = this.memories.find(m => m.id === memoryId);
        if (!memory) {
            throw new Error('Memory not found');
        }
        
        // 全員で共有するために、投影モードで再生
        this.project(memory);
    }
    
    project(memory) {
        const projector = new MemoryProjector();
        projector.setRange(10); // 10m範囲
        projector.setMode('shared'); // 共有モード
        projector.play(memory);
    }
}

class EnvironmentalAnalyzer {
    constructor() {
        this.sensors = new SensorArray();
        this.history = [];
    }
    
    measureVitality() {
        const soil = this.sensors.readSoil();
        const air = this.sensors.readAir();
        const flora = this.sensors.scanFlora();
        
        return {
            soilHealth: soil.nutrients / soil.magicDensity,
            airQuality: air.oxygen / air.magicParticles,
            biodiversity: flora.speciesCount / EXPECTED_SPECIES_COUNT
        };
    }
    
    calculateDecline() {
        const current = this.measureVitality();
        
        // 三百年前と比較
        const baseline = this.history.getBaseline(300);
        
        return {
            soilDecline: (baseline.soilHealth - current.soilHealth) / baseline.soilHealth,
            airDecline: (baseline.airQuality - current.airQuality) / baseline.airQuality,
            biodiversityDecline: (baseline.biodiversity - current.biodiversity) / baseline.biodiversity
        };
    }
    
    predictFuture() {
        const decline = this.calculateDecline();
        const yearsUntilCollapse = [];
        
        for (let aspect in decline) {
            const rate = decline[aspect];
            const remaining = 1 - rate;
            const years = remaining / rate * 300;
            yearsUntilCollapse.push(years);
        }
        
        return Math.min(...yearsUntilCollapse);
    }
}

class AncientTextDecoder {
    constructor() {
        this.dictionary = this.loadDictionary();
        this.patterns = this.identifyPatterns();
    }
    
    decode(text) {
        let decoded = '';
        
        for (let char of text) {
            const mapping = this.dictionary.find(char);
            if (mapping) {
                decoded += mapping.translation;
            } else {
                decoded += this.inferMeaning(char);
            }
        }
        
        return this.structure(decoded);
    }
    
    inferMeaning(char) {
        // 文脈から意味を推測
        const context = this.analyzeContext();
        const similar = this.findSimilarChars(char);
        
        return this.bestGuess(similar, context);
    }
    
    structure(text) {
        // 構造化された知識として整理
        return {
            original: text,
            summary: this.summarize(text),
            keyPoints: this.extractKeyPoints(text),
            warnings: this.extractWarnings(text)
        };
    }
}

class HarmonyAttunement {
    constructor() {
        this.resonance = 0;
        this.targetResonance = HARMONY_THRESHOLD;
    }
    
    synchronize(crystal) {
        const crystalFreq = crystal.getFrequency();
        const myFreq = this.getMyFrequency();
        
        // 周波数を調整
        const adjustment = this.calculateAdjustment(crystalFreq, myFreq);
        this.applyAdjustment(adjustment);
        
        return this.measureResonance(crystal);
    }
    
    measureResonance(crystal) {
        const crystalFreq = crystal.getFrequency();
        const myFreq = this.getMyFrequency();
        
        const difference = Math.abs(crystalFreq - myFreq);
        const maxFreq = Math.max(crystalFreq, myFreq);
        
        this.resonance = 1 - (difference / maxFreq);
        return this.resonance;
    }
    
    isHarmonized() {
        return this.resonance >= this.targetResonance;
    }
}
