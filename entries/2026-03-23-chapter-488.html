// イグニス戦力分析
class IgnisAnalyzer {
    constructor() {
        this.base = {
            name: "Ignis",
            title: "Fire Sage",
            element: "fire",
            basePower: 90
        };
        this.buffs = [
            {name: "LindManaBoost", multiplier: 1.5, duration: "permanent"},
            {name: "TerritoryBonus", multiplier: 1.3, duration: "in_fire"}
        ];
    }

    calculateCurrentPower() {
        let power = this.base.basePower;
        this.buffs.forEach(b => power *= b.multiplier);
        return Math.round(power); // 約175
    }
}

const analyzer = new IgnisAnalyzer();
const ignisPower = analyzer.calculateCurrentPower();
