    function restorePhysicalDamage(floor) {
      // 地の精霊を召喚
      const earth = summonElemental("earth");
      
      // 損傷箇所を特定
      const damages = scan(floor, "physical_damage");
      
      for (const damage of damages) {
        // 元の状態を復元
        const originalState = analyzeOriginal(damage);
        
        // 石材を再結合
        earth.reconstruct({
          target: damage,
          blueprint: originalState,
          speed: "moderate"
        });
      }
