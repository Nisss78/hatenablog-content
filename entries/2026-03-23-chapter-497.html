// Recovery Phase
class PostBattleRecovery {
    constructor(party) {
        this.party = party;
        this.status = {
            kaito: { hp: 85, mp: 60, status: "normal" },
            riamu: { hp: 70, mp: 90, status: "軽傷" },
            erina: { hp: 90, mp: 75, status: "normal" },
            mina: { hp: 95, mp: 40, status: "魔力枯渇" },
            cecilia: { hp: 100, mp: 85, status: "normal" },
            teradon: { hp: 40, mp: 20, status: "疲労" }
        };
    }
    
    healAll() {
        Object.keys(this.status).forEach(member => {
            this.status[member].hp = 100;
            this.status[member].mp = 100;
            this.status[member].status = "normal";
        });
        return {
            result: "全員回復完了",
            duration: "約10分"
        };
    }
}

// Gravity Manipulation Code
class GravityManipulation {
    constructor(target) {
        this.target = target;
        this.mass = target.mass; // kg
        this.currentGravity = 9.8; // m/s²
        this.manipulatedGravity = 0;
    }
    
    reduceGravity(factor) {
        this.manipulatedGravity = this.currentGravity / factor;
        return {
            newGravity: this.manipulatedGravity,
            forceReduction: `${(1 - 1/factor) * 100}%`
        };
    }
    
    levitate() {
        this.manipulatedGravity = 0;
        return {
            status: "levitating",
            requiredControl: "constant_mp_flow",
            strain: "high"
        };
    }
    
    control(direction, force) {
        return {
            action: "move",
            direction: direction,
            appliedForce: force,
            mpCost: force * 2
        };
    }
}

// Guardian's Will Analysis
class GuardianWill {
    constructor() {
        this.protected = [];
        this.willStrength = 0;
    }
    
    analyzeHeart(hero) {
        const feelings = hero.hearth.feelings;
        const important = feelings.filter(f => f.type === "protection");
        
        important.forEach(person => {
            this.protected.push({
                name: person.name,
                bondStrength: person.bond,
                reason: person.reason
            });
        });
        
        this.willStrength = this.calculateWill();
        return {
            protectedCount: this.protected.length,
            willType: this.determineWillType(),
            strength: this.willStrength
        };
    }
    
    determineWillType() {
        if (this.willStrength > 80) return "絶対の守護";
        if (this.willStrength > 50) return "強き守護";
        return "萌芽の守護";
    }
    
    calculateWill() {
        let total = 0;
        this.protected.forEach(p => {
            total += p.bondStrength;
        });
        return Math.min(100, total);
    }
}

const will = new GuardianWill();
const result = will.analyzeHeart(kaito);
// result: { protectedCount: 50+, willType: "絶対の守護", strength: 95 }
