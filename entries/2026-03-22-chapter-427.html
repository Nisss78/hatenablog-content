// Fire Manifestation
// 火の顕現

function createFire(intensity, duration) {
    const mana = gatherMana(intensity);
    const flame = manifest(mana, "fire");
    return {
        element: "fire",
        temperature: intensity * 100,
        duration: duration,
        stability: stabilize(flame)
    };
}
