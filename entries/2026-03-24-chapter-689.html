class ThermalAnalysis:
    def __init__(self):
        self.base_temp = 25  # 摂氏
        self.current_temp = 0
        self.heat_source = None
    
    def measure_ambient(self):
        readings = self.collect_sensor_data()
        self.current_temp = sum(readings) / len(readings)
        return self.current_temp
    
    def locate_heat_source(self):
        gradients = []
        for direction in ["north", "south", "east", "west", "up", "down"]:
            gradients.append((direction, self.get_temp_gradient(direction)))
        
        return max(gradients, key=lambda x: x[1])[0]
    
    def estimate_distance(self):
        # 逆二乗則で熱源までの距離を推定
        ratio = self.current_temp / self.base_temp
        if ratio > 1:
            return f"熱源は近い（温度上昇率: {ratio:.2f}）"
        return "熱源は遠い"

class FlameBarrier:
    def __init__(self):
        self.temperature = 1200  # 摂氏
        self.magical_nature = "ENCHANTED_FIRE"
        self.weakness = None
    
    def analyze_structure(self):
        # 炎の構造を解析
        patterns = self.detect_magic_patterns()
        if patterns:
            return "魔法で維持されている炎"
        return "自然発火"
    
    def find_extinguish_method(self):
        methods = ["water", "ice", "void", "countermagic"]
        for method in methods:
            if self.test_effectiveness(method) > 0.7:
                self.weakness = method
                return method
        return None

def reinforce_magic(primary, secondary):
    # primary: シオンの火魔法
    # secondary: カイトの虚無魔法
    
    combined = MagicWeave()
    combined.core = primary.flame_core()
