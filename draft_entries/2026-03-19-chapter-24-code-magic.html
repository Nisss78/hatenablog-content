class CrystalOfPromises {
  constructor() {
    this.worlds = new Map();
    this.keys = new Map();
  }
  
  registerWorld(worldId, key) {
    this.worlds.set(worldId, true);
    this.keys.set(worldId, key);
    console.log(`Registered: ${worldId}`);
  }
  
  async converge() {
    // Check if all worlds are registered
    if (this.worlds.size !== 3) {
      throw new Error('Not all worlds are registered');
    }
    
    // Gather all keys
    const keys = Array.from(this.keys.values());
    
    // Start convergence
    return this.executeConvergence(keys);
  }
  
  async executeConvergence(keys) {
    const promises = keys.map((key, index) => {
      return this.syncWorld(index, key);
    });
    
    return Promise.all(promises);
  }
}

// World Convergence Sequence
async function startConvergence(crystal) {
  try {
    console.log('Starting world convergence...');
    
    // Phase 1: Synchronize all worlds
    await crystal.synchronizeWorlds();
    console.log('✓ Synchronization complete');
    
    // Phase 2: Merge memories
    await crystal.mergeMemories();
    console.log('✓ Memory merge complete');
    
    // Phase 3: Unify timeline
    await crystal.unifyTimeline();
